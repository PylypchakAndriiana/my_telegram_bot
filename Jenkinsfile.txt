pipeline {
    agent any

    environment {
        PYTHONUNBUFFERED = '1'
        PYTHON_PATH = "C:\\Users\\User\\AppData\\Local\\Programs\\Python\\Python313\\python.exe"
        VENV_PYTHON = "venv\\Scripts\\python.exe"
    }

    stages {
        stage('Створення віртуального середовища') {
            steps {
                script {
                    if (!fileExists('venv')) {
                        echo 'Створення віртуального середовища...'
                        bat """
                            ${PYTHON_PATH} -m venv venv
                        """
                    } else {
                        echo 'Віртуальне середовище вже існує.'
                    }
                }
            }
        }

        stage('Встановлення залежностей') {
            steps {
                echo 'Встановлення залежностей...'
                bat '''
                    venv\\Scripts\\python.exe -m pip install --upgrade pip
                    venv\\Scripts\\python.exe -m pip install -r requirements.txt
                '''
            }
        }

        stage('Видалення вебхука Telegram') {
            steps {
                echo 'Видалення вебхука Telegram перед запуском polling...'
                bat '''
                    venv\\Scripts\\python.exe -c "import telebot; bot = telebot.TeleBot('TOKEN'); bot.delete_webhook()"
                '''
            }
        }

        stage('Запуск бота') {
            steps {
                echo 'Запуск бота...'
                bat 'venv\\Scripts\\python.exe bot.py'
            }
        }
    }

    post {
        always {
            echo 'Процес завершено.'
        }
    }
}
