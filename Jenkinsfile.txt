pipeline {
    agent any

    environment {
        PYTHONUNBUFFERED = '1'
        VENV_PATH = 'venv\\Scripts\\activate'  // Шлях до активації віртуального середовища
        PYTHON_PATH = "C:\\Users\\User\\AppData\\Local\\Programs\\Python\\Python313\\python.exe"
    }

    stages {
        stage('Створення віртуального середовища') {
            steps {
                script {
                    // Якщо віртуальне середовище вже існує, пропускаємо створення
                    if (!fileExists('venv')) {
                        bat """
                            ${PYTHON_PATH} -m venv venv
                            call venv\\Scripts\\activate
                            pip install --upgrade pip
                            pip install -r requirements.txt
                        """
                    } else {
                        echo 'Віртуальне середовище вже існує.'
                    }
                }
            }
        }

        stage('Запуск бота') {
            steps {
                bat '''
                    call venv\\Scripts\\activate
                    python bot.py
                '''
            }
        }
    }
}
